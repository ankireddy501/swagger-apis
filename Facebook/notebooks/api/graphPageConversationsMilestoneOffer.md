```javascript

function getDate(){

  var date = new Date()

  var h = date.getHours()

  var min = date.getMinutes()

  var m = date.getMonth() + 1

  var d = date.getDate()

  var y = date.getFullYear()

  var s = date.getSeconds()

  var result = m+'.'+d+'.'+y+'/'+h+':'+min+':'+s

  return result

}
```
